import { Suspense, lazy } from 'react';
const Spline = lazy(() => import('@splinetool/react-spline'));

interface InteractiveRobotSplineProps {
  scene: string;
  className?: string;
}

export function InteractiveRobotSpline({ scene, className }: InteractiveRobotSplineProps) {
  return (
    <Suspense
      fallback={

          </svg>
        </div>
      }
    >
      
      <Spline
        scene={scene}
        className={className} 
      />
    </Suspense>
  );
